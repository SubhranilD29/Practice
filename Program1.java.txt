public class Program1 {
    public static void main(String[] args) {

        int num1 = 50;
        int num2 = 30;


        System.out.println("Before swapping:");
        System.out.println("num1 = " + num1);
        System.out.println("num2 = " + num2);


        num1=num1+num2;
	num2=num1-num2;
	num1=num1-num2;


        System.out.println("After swapping:");
        System.out.println("num1 = " + num1);
        System.out.println("num2 = " + num2);
    }
}
